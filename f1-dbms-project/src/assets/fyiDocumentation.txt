Stored Procedures
1. GetDriverStandings

Purpose: Returns season-wise driver standings ordered by total points.

SQL:

DELIMITER //


CREATE PROCEDURE GetDriverStandings(
IN seasonYear YEAR
)


BEGIN


SELECT
    d.FirstName,
    d.LastName,
    SUM(r.Points) AS Points
FROM RESULTS r


JOIN DRIVERS d
ON r.DriverID=d.DriverID


JOIN RACES ra
ON r.RaceID=ra.RaceID


WHERE ra.Season=seasonYear


GROUP BY d.DriverID


ORDER BY Points DESC;


END //


DELIMITER ;

Usage:

CALL GetDriverStandings(2026);
2. GetTeamStandings

Purpose: Returns constructor standings for a selected season.

SQL:

DELIMITER //


CREATE PROCEDURE GetTeamStandings(
IN seasonYear YEAR
)


BEGIN


SELECT
    st.DisplayName,
    SUM(r.Points) AS Points
FROM RESULTS r


JOIN RACES ra
ON r.RaceID=ra.RaceID


JOIN SEASON_TEAMS st
ON r.TeamID=st.TeamID
AND st.Season=ra.Season


WHERE ra.Season=seasonYear


GROUP BY st.TeamID


ORDER BY Points DESC;


END //


DELIMITER ;

Usage:

CALL GetTeamStandings(2026);
